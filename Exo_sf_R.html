A. Import des données

Lister les couches géographiques d’un fichier GeoPackage

st_layers(“data/GeoSenegal.gpkg”)

Import des données géographiques

localites <- st_read(dsn = “data/GeoSenegal.gpkg”,layer = “Localites” ) Pays_voisins <- st_read(dsn=“data/GeoSenegal.gpkg”,layer = “Pays_voisins”) sen <- st_read(dsn=“data/GeoSenegal.gpkg”,layer = “Senegal”) Departements <- st_read(dsn=“data/GeoSenegal.gpkg”,layer = “Departements”) USSEIN <- st_read(dsn=“data/GeoSenegal.gpkg”,layer = “USSEIN”) Routes <- st_read(dsn=“data/GeoSenegal.gpkg”,layer = “Routes”)

B. Séléction et intersection spatiale

loc_sn_11 <- subset(localites, PAYS == “SN”)

Solution 2 - par localisation

loc_sn_2 <- st_filter(x = localites, y= sen, predicate = st_intersects)
